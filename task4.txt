A = int(input("количество лет, в течение которых посетитель будет смотреть на экспонаты: "))
B = (8*60)/5
print(365*A*B, "экспонатов будет просмотрено за", A, "лет")

C = int(input("количество экспонатов: "))
G = C*5
Y = G//(365*8*60)
D = (G - Y*365*480)//480
M = G - Y*365*480 - D*480
print("времени будет потрачено:", Y, "лет", D, "дней", M, "минут")
